#include "CControl/ccontrol.h"

#define RDIM 1
#define LIMIT 1
#define LEARNING 0.001

int main() {

	/* MRAC works as K1 and K2 integrates over time so y -> r when time -> inf
	 * r ----{K1}*(-LEARNING)---->+()----->G(s)-----------------------> y
	 *               			  -^            			  |
	 *               			   |            			  |
	 *                			   |              			  |
	 *                  		   |---*(LEARNING){K2}<--------
	 */

	float y[RDIM] = {7};
	float u[RDIM] = {5};
	float r[RDIM] = {51};
	float I1[RDIM] = {3.1};
	float I2[RDIM] = {4.1};

	/* Compute now */
	mrac(LIMIT, LEARNING, y, u, r, I1, I2, RDIM);

	/* Show input signal */
	printf("Our output signal\n");
	print(u, RDIM, 1);

	return EXIT_SUCCESS;
}
