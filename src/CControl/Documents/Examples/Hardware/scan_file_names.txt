#include "CControl/Headers/Functions.h"

int main() {
	clock_t start, end;
	float cpu_time_used;
	start = clock();

	/*
	 * Start counting files and its names
	 * Enter your folder path, in this case, it's Windows folder path for me
	 */
	char** file_names = NULL;
	size_t sub_folder_count = scan_file_names("C:\\Users\\dmn\\VS", &file_names);

	/* Print the files */
	size_t i;
	for (i = 0; i < sub_folder_count; i++) {
		printf("Mapp %d: %s\n", i + 1, file_names[i]);
	}

	/* Free */
	for (i = 0; i < sub_folder_count; i++) {
		free(file_names[i]);
	}
	free(file_names);

	end = clock();
	cpu_time_used = ((float)(end - start)) / CLOCKS_PER_SEC;
	printf("\nTotal speed  was %f\n", cpu_time_used);

	return EXIT_SUCCESS;
}
