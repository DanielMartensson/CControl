#include "CControl/ccontrol.h"

#define row 5
#define column 4

int main() {
	clock_t start, end;
	float cpu_time_used;
	start = clock();

	/* Matrix */
	float A[row * column] = { 3,    3,    5,    6,
							  2,    3,    4,    5,
							  5,    4,   65,    0,
							  1,    4,    6,    1,
							- 5,    2,    6,   10 };

	/* Compute the rank */
	printf("The rank of matrix A is: %i\n", rank(A, row, column));

	end = clock();
	cpu_time_used = ((float)(end - start)) / CLOCKS_PER_SEC;
	printf("\nTotal speed  was %f\n", cpu_time_used);
	return EXIT_SUCCESS;
}
