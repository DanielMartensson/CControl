#include "CControl/Headers/Functions.h"

int main() {
	// Start clock
	clock_t start, end;
	float cpu_time_used;
	start = clock();

	float x[1000];
	randn(x, 1000, 10, 10);
	float Mean = 0;
	for(int i = 0; i < 1000; i++)
		Mean += x[i];
	Mean = Mean/1000;
	printf("Mean = %f\n", Mean);


	// Compute the speed
	end = clock();
	cpu_time_used = ((float) (end - start)) / CLOCKS_PER_SEC;
	printf("\nTotal speed  was %f\n", cpu_time_used);

	return EXIT_SUCCESS;
}
