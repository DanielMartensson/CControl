#include "CControl/ccontrol.h"

#define row_a 7
#define row_b 5
#define column_a 1
#define column_b 1

int main() {
	clock_t start, end;
	float cpu_time_used;
	start = clock();

	/* Define A, B and C */
	float A[row_a*column_a] = { 0.5723,   0.1744,   0.4915, - 0.6698, - 0.5433,   1.5792, - 0.4664};
	float B[row_b*column_b] = { 4.1830,   5.3727,   4.6562,   4.4651,   4.7539 };
	float C[row_a * row_b];

	/* Compute pdist2 */
	pdist2(A, B, C, row_a, column_a, row_b, PDIST2_METRIC_L2);

	/* Print C */
	print(C, row_a, row_b);

	end = clock();
	cpu_time_used = ((float) (end - start)) / CLOCKS_PER_SEC;
	printf("\nTotal speed  was %f\n", cpu_time_used);

	return EXIT_SUCCESS;
}
