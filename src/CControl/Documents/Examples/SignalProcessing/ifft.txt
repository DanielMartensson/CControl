#include "CControl/ccontrol.h"

#define n 11

int main() {
	clock_t start, end;
	float cpu_time_used;
	start = clock();

	/* Create x */
	float x[n] = {1, 2, 5, 10, 5, 3.5, 1, 2, 3, 10, 2};

	/* Do Inverse FFT */
	fft(x, n);
	ifft(x, n);

	/* Print */
	print(x, 1, n);

	end = clock();
	cpu_time_used = ((float)(end - start)) / CLOCKS_PER_SEC;
	printf("\nTotal speed  was %f\n", cpu_time_used);

	return EXIT_SUCCESS;
}
