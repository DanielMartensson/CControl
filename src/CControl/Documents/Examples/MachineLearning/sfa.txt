#include "CControl/ccontrol.h"

int main() {
	clock_t start, end;
	float cpu_time_used;
	start = clock();

	/* Read image */
	PGM* image = imread("..\\src\\CControl\\Documents\\Data\\fast\\lab.pgm");

	/* Create matrix X */
	const size_t row = image->height;
	const size_t column = image->width;
	float* X = (float*)malloc(row * column * sizeof(float));
	size_t i, j;
	for (i = 0; i < row * column; i++) {
		X[i] = (float)image->pixels[i];
	}

	/* Tuning parameters */
	FAST_METHOD fast_method = FAST_METHOD_9;
	int fast_threshold = 50;
	uint8_t sobel_threshold = 250;
	uint8_t histogram_size = 10;

	/* Find histogram by using SFA */
	float* histogram = sfa(X, fast_threshold, sobel_threshold, fast_method, histogram_size, row, column);
	end = clock();

	/* Print histogram */
	print(histogram, histogram_size, 1);

	/* Free */
	free(X);
	imfree(image);
	free(histogram);

	/* Detect memory leakage */
	detectmemoryleak();

	cpu_time_used = ((float)(end - start)) / CLOCKS_PER_SEC;
	printf("\nTotal speed  was %f\n", cpu_time_used);

	return EXIT_SUCCESS;
}
